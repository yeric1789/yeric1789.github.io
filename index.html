# Write SQL query
winning_p_query = """ WITH sal(teamID, yearID,salary) AS
                      (SELECT teamID, yearID, sum(salary)
                      FROM Salaries
                      GROUP BY teamID, yearID),
                      t(teamID,yearID,lgID,franchID,rank,W,G,w_p) AS
                      (SELECT Teams.teamID, Teams.yearID,Teams.lgID, Teams.franchID, Teams.rank, Teams.W,Teams.G, ((Teams.W*1.0/Teams.G)*100)
                      FROM Teams
                      GROUP BY teamID, yearID)
                      SELECT sal.teamID, sal.yearID,salary,lgID,franchID,rank,W,G,w_p
                      FROM sal,t
                      WHERE
                      t.teamID == sal.teamID AND t.yearID == sal.yearID
                      """
df = pandas.read_sql(winning_p_query, conn)

Now we will compute the effeciency by using the given equation: $efficiency_{ij} = win\_pct_{ij}-expected\_win\_pct_{ij}$. Also note that now the f functionality will come in handy.

In [50]:
# First lets recreate the dataframe with only the years without intervals:
data = df[df["yearID"] >= 1990].copy()
data = data.groupby(["yearID","teamID"]).mean().reset_index(level=['yearID', 'teamID'])
mean_salary = data.groupby(["yearID"])["salary"].mean().to_frame()
mean_salary["Std"] = data.groupby(["yearID"])["salary"].std()
data["Standardized_salary"] = data.apply(lambda x: (x["salary"]-mean_salary.loc[x["yearID"]]["salary"])/mean_salary.loc[x["yearID"]]["Std"], axis = 1)

#Define anonymous function 
ef = lambda x: x["w_p"]-f(x["Standardized_salary"])
data["efficiency"] = data.apply(ef,axis = 1)
